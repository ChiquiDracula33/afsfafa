        let currentQuestion = 0;
        let userData = {};

        // Iniciar cuestionario
        document.getElementById('startQuiz').addEventListener('click', function() {
            userData = {
                name: document.getElementById('userName').value,
                birthDate: document.getElementById('birthDate').value,
                education: document.getElementById('educationLevel').value
            };
            
            if (!userData.name) {
                alert("Por favor ingresa tu nombre");
                return;
            }
            
            document.getElementById('startScreen').style.display = 'none';
            document.getElementById('quizScreen').style.display = 'block';
            document.getElementById('userGreeting').textContent = `Hola ${userData.name}, ¡comencemos!`;
            loadQuestion();
        });

        function loadQuestion() {
            const questionElement = document.getElementById('question');
            const optionsElement = document.getElementById('options');
            const nextButton = document.getElementById('next');

            if (currentQuestion < questions.length) {
                questionElement.textContent = questions[currentQuestion].question;
                optionsElement.innerHTML = '';
                questions[currentQuestion].options.forEach((option, index) => {
                    const button = document.createElement('button');
                    button.textContent = option;
                    button.className = 'option';
                    button.onclick = () => checkAnswer(index);
                    optionsElement.appendChild(button);
                });
                nextButton.style.display = 'none';
            } else {
                // Mostrar pantalla final
                document.getElementById('quizScreen').style.display = 'none';
                const endScreen = document.getElementById('endScreen');
                endScreen.style.display = 'block';
                document.getElementById('finalMessage').innerHTML = `
                    <h2>¡Felicidades ${userData.name}!</h2>
                    <p style="font-size: 1.2em;">Has completado el test de Inteligencia Financiera.</p>
                    <p style="font-size: 1.1em;">Gracias por participar y esperamos que hayas aprendido conceptos valiosos.</p>
                `;
                createConfetti();
            }
        }

        function checkAnswer(selectedIndex) {
            const modal = document.getElementById('myModal');
            const modalHeader = document.getElementById('modalHeader');
            const modalTitle = document.getElementById('modalTitle');
            const modalFeedback = document.getElementById('modalFeedback');
            const correctAnswer = questions[currentQuestion].answer;

            if (selectedIndex === correctAnswer) {
                modalHeader.className = 'modal-header correct';
                const messages = [`¡CORRECTO ${userData.name.toUpperCase()}! 🎉`, "¡EXCELENTE! 👏", "¡MUY BIEN! 💡", "¡ERES INCREÍBLE! 🚀"];
                const randomMessage = messages[Math.floor(Math.random() * messages.length)];
                modalTitle.textContent = randomMessage;
                modalFeedback.innerHTML = `<p style="font-size: 1.3em; margin-top: 20px;">${questions[currentQuestion].explanation}</p>`;
                createConfetti();
            } else {
                modalHeader.className = 'modal-header incorrect';
                modalTitle.textContent = "¡INCORRECTO! 😕";
                modalFeedback.innerHTML = `
                    <p style="font-weight: bold; font-size: 1.2em;">La respuesta correcta es: ${questions[currentQuestion].options[correctAnswer]}</p>
                    <p style="font-size: 1.3em; margin-top: 20px;">${questions[currentQuestion].explanation}</p>
                `;
            }
            modal.style.display = "block";
        }

        function closeModal() {
            const modal = document.getElementById('myModal');
            modal.style.display = "none";
            currentQuestion++;
            loadQuestion();
        }

        function nextQuestion() {
            closeModal();
        }

        function createConfetti() {
            const container = document.querySelector('.container');
            for (let i = 0; i < 50; i++) {
                const confetti = document.createElement('div');
                confetti.className = 'confetti';
                confetti.style.left = Math.random() * 100 + '%';
                confetti.style.backgroundColor = getRandomColor();
                confetti.style.width = Math.random() * 10 + 5 + 'px';
                confetti.style.height = confetti.style.width;
                confetti.style.animationDuration = Math.random() * 3 + 2 + 's';
                container.appendChild(confetti);
                
                // Eliminar el confeti después de la animación
                setTimeout(() => {
                    confetti.remove();
                }, 5000);
            }
        }

        function getRandomColor() {
            const colors = ['#2563EB', '#F59E0B', '#10B981', '#3B82F6', '#EF4444', '#8B5CF6'];
            return colors[Math.floor(Math.random() * colors.length)];
        }
